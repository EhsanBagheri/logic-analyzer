107  inline void write(T value){
108  if (ignore_count > 0) {
109  ignore_count--;
110  return;
111  }
112  data[write_pos] = value;
113  write_pos++;
114  if (write_pos>size_count){
115  write_pos = 0;
116  }
117  available_count++;
118  if (available_count>size_count){
119  available_count = size_count;
120  }
121  }
122 
124  T read() {
125  T result = 0;
126  if (available_count>0){
127  result = data[read_pos++];
128  if (read_pos>size_count){
129  read_pos = 0;
130  }
131  available_count--;
132  }
133  return result;
134  }
135 
137  void clear() {
138  available_count = 0;
139  write_pos = 0;
140  read_pos = 0;
141  }
142 
144  void clear(int count){
145  if (count>available_count){
146  ignore_count = count - available_count;
147  size_count = available_count;
148  }
149  for (int j=0;j<count;j++){
150  read();
151  }
152  }
153 
155  inline int available() {
156  return available_count;
157  }
158 
160  int size() {
161  return size_count;
162  }
163 
164  T *data_ptr(){
165  return data;
166  }
167

169  int available_count = 0;
170  int size_count = 0;
171  int write_pos = 0;
172  int read_pos = 0;
173  int ignore_count = 0;

236  virtual void capture(bool is_max_speed=false, bool is_dump=true) {
237  printLog("capture(trigger)");
238  // waitring for trigger
239  if (trigger_mask) {
240  printLog("waiting for trigger");
241  while ((trigger_values ^ captureSample()) & trigger_mask)
242  ;
243  printLog("triggered");
244  setStatus(TRIGGERED);
245  } else {
246  setStatus(TRIGGERED);
247  }
248 
249  // remove unnecessary entries from buffer based on delayCount & readCount
250  printLog("capture(buffer)");
251  long keep = read_count - delay_count;
252  if (keep > 0 && buffer_ptr->available()>keep) {
253  printLog("keeping last %ld entries",keep);
254  buffer_ptr->clear(buffer_ptr->available() - keep);
255  } else if (keep < 0) {
256  printLog("ignoring first %ld entries",abs(keep));
257  buffer_ptr->clear(buffer_ptr->available() + abs(keep));
258  } else if (keep==0l){
259  printLog("starting with clean buffer");
260  buffer_ptr->clear();
261  }
262 
263  // Capture
264  if (is_max_speed){
265  if (is_continuous_capture){
266  captureAllContinousMaxSpeed();
267  } else {
268  captureAllMaxSpeed();
269  setStatus(STOPPED);
270  printLog("capture-done: %lu",buffer_ptr->available());

600  void processCommand(int cmd){
601  if (buffer_ptr==nullptr || impl_ptr==nullptr) return;
602 
603  switch (cmd) {
607  case SUMP_RESET:
608  // debounce reset
609  if (millis()>sump_reset_igorne_timeout){
610  printLog("->SUMP_RESET");
611  sump_reset_igorne_timeout = millis()+ 500;
612  }
613  break;
614 
618  case SUMP_ID:
619  printLog("->SUMP_ID");
620  this->stream().write(device_id, 4);
621  this->stream().flush();
622  break;
623 
624  /*
625  * Captures the data
626  */
627  case SUMP_ARM:
628  printLog("->SUMP_ARM");
629  setStatus(ARMED);
630  capture(frequecy_value >= max_frequecy_value); // if frequecy_value >= max_frequecy_value -> capture at max speed
631  break;
632 
633  /*
634  * the trigger mask byte has a '1' for each enabled trigger so
635  * we can just use it directly as our trigger mask.
636  */
637  case SUMP_TRIGGER_MASK:
638  printLog("->SUMP_TRIGGER_MASK");
639  setTriggerMask(commandExtPinBitArray());
640  break;
641 
642  /*
643  * trigger_values can be used directly as the value of each bit
644  * defines whether we're looking for it to be high or low.
645  */
646  case SUMP_TRIGGER_VALUES:
647  printLog("->SUMP_TRIGGER_VALUES");
648  setTriggerValues(commandExtPinBitArray());
649  break;
650 
651 
652  /* read the rest of the command bytes but ignore them. */
653  case SUMP_TRIGGER_CONFIG:
654  printLog("->SUMP_TRIGGER_CONFIG");
655  getSump4ByteComandArg();
656  break;
657  /*
658  * the shifting needs to be done on the 32bit unsigned long variable
659  * so that << 16 doesn't end up as zero.
660  */
661  case SUMP_SET_DIVIDER: {
662  printLog("->SUMP_SET_DIVIDER");
663  Sump4ByteComandArg cmd = getSump4ByteComandArg();
664  uint32_t divider = cmd.get32();
665  printLog("-divider: %lu\n", divider);
666  setupDelay(divider);
667  }
668  break;
669 
670  /*
671  * this just sets up how many samples there should be before
672  * and after the trigger fires. The readCount is total samples
673  * to return and delayCount number of samples after the trigger.
674  * this sets the buffer splits like 0/100, 25/75, 50/50
675  * for example if readCount == delayCount then we should
676  * return all samples starting from the trigger point.
677  * if delayCount < readCount we return (readCount - delayCount) of
678  * samples from before the trigger fired.
679  */
680  case SUMP_SET_READ_DELAY_COUNT: {
681  printLog("->SUMP_SET_READ_DELAY_COUNT");
682  Sump4ByteComandArg cmd = getSump4ByteComandArg();
683  read_count = min((uint32_t)cmd.get16(0), max_capture_size);
684  delay_count = min((uint32_t)cmd.get16(1),max_capture_size);
685  printLog("--> read_count: %d", read_count);
686  printLog("--> delay_count: %d", delay_count);
687  raiseEvent(READ_DLEAY_COUNT);
688  }
689  break;
690 
691  /* read the rest of the command bytes and check if RLE is enabled. */
692  case SUMP_SET_FLAGS: {
693  printLog("->SUMP_SET_FLAGS");
694  Sump4ByteComandArg cmd = getSump4ByteComandArg();
695  is_continuous_capture = ((cmd.getPtr()[1] & 0B1000000) != 0);
696  printLog("--> is_continuous_capture: %d\n", is_continuous_capture);
697  raiseEvent(FLAGS);
698 
699  }
700  break;
701 
702  /*
703  * We return a description of our capabilities.
704  * Check the function's comments below.
705  */
706  case SUMP_GET_METADATA:
707  printLog("->SUMP_GET_METADATA");
708  sendMetadata();
709  break;
710 
711  /* ignore any unrecognized bytes. */
712  default:
713  printLog("->UNHANDLED command: %d", cmd);
714  break;
715 
716  };
717  }
