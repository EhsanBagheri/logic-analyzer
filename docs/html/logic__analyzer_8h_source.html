137  available_count--;

382  // actual state
383  PinBitArray actual = pinReader().readAll();
384 
385  // buffer single capture cycle
386  if (state_ptr->is_continuous_capture) {
387  write(actual);

396  void capture(bool is_max_speed) {
397  printLog("capture(trigger)");
398  // waiting for trigger
399  if (state_ptr->trigger_mask) {
400  printLog("waiting for trigger");
401  while ((state_ptr->trigger_values ^ captureSample()) & state_ptr->trigger_mask)
402  ;
403  printLog("triggered");
404  setStatus(TRIGGERED);
405  } else {
406  setStatus(TRIGGERED);
407  }
408 
409  // remove unnecessary entries from buffer based on delayCount & readCount
410  printLog("capture(buffer)");
411  long keep = state_ptr->read_count - state_ptr->delay_count;
412  if (keep > 0 && buffer_ptr->available()>keep) {
413  printLog("keeping last %ld entries",keep);
414  buffer_ptr->clear(buffer_ptr->available() - keep);
415  } else if (keep < 0) {
416  printLog("ignoring first %ld entries",abs(keep));
417  buffer_ptr->clear(buffer_ptr->available() + abs(keep));
418  } else if (keep==0l){
419  printLog("starting with clean buffer");
420  buffer_ptr->clear();
421  }
422 
423  // Start Capture
424  if (is_max_speed){
425  if (state_ptr->is_continuous_capture){
426  captureAllContinousMaxSpeed();
427  } else {
428  captureAllMaxSpeed();
429  setStatus(STOPPED);
430  printLog("capture-done: %lu",buffer_ptr->available());
431  dumpData();
432  }
433  } else {
434  if (state_ptr->is_continuous_capture){
435  captureAllContinous();
436  } else {
437  captureAll();
438  setStatus(STOPPED);
439  printLog("capture-done: %lu",buffer_ptr->available());
440  dumpData();
441  }
442  }
443  }

768  void processCommand(int cmd){
769 
770  switch (cmd) {
774  case SUMP_RESET:
775  // debounce reset
776  if (millis()>sump_reset_igorne_timeout){
777  printLog("->SUMP_RESET");
778  sump_reset_igorne_timeout = millis()+ 500;
779  }
780  break;
781 
785  case SUMP_ID:
786  printLog("->SUMP_ID");
787  this->stream().write(device_id, 4);
788  this->stream().flush();
789  break;
790 
791  /*
792  * We return a description of our capabilities.
793  * Check the function's comments below.
794  */
795  case SUMP_GET_METADATA:
796  printLog("->SUMP_GET_METADATA");
797  sendMetadata();
798  break;
799 
800  /*
801  * Captures the data
802  */
803  case SUMP_ARM:
804  printLog("->SUMP_ARM");
805  setStatus(ARMED);
806  if (capture_ptr!=nullptr && is_capture_on_arm){
807  capture_ptr->capture();
808  }
809  break;
810 
811  /*
812  * the trigger mask byte has a '1' for each enabled trigger so
813  * we can just use it directly as our trigger mask.
814  */
815  case SUMP_TRIGGER_MASK:
816  printLog("->SUMP_TRIGGER_MASK");
817  setTriggerMask(commandExtPinBitArray());
818  break;
819 
820  /*
821  * trigger_values can be used directly as the value of each bit
822  * defines whether we're looking for it to be high or low.
823  */
824  case SUMP_TRIGGER_VALUES:
825  printLog("->SUMP_TRIGGER_VALUES");
826  setTriggerValues(commandExtPinBitArray());
827  break;
828 
829 
830  /* read the rest of the command bytes but ignore them. */
831  case SUMP_TRIGGER_CONFIG:
832  printLog("->SUMP_TRIGGER_CONFIG");
833  getSump4ByteComandArg();
834  break;
835 
836  /*
837  * the shifting needs to be done on the 32bit unsigned long variable
838  * so that << 16 doesn't end up as zero.
839  */
840  case SUMP_SET_DIVIDER: {
841  printLog("->SUMP_SET_DIVIDER");
842  Sump4ByteComandArg cmd = getSump4ByteComandArg();
843  uint32_t divider = cmd.get32();
844  printLog("-divider: %lu\n", divider);
845  setupDelay(divider);
846  }
847  break;
848 
849  /*
850  * this just sets up how many samples there should be before
851  * and after the trigger fires. The readCount is total samples
852  * to return and delayCount number of samples after the trigger.
853  * this sets the buffer splits like 0/100, 25/75, 50/50
854  * for example if readCount == delayCount then we should
855  * return all samples starting from the trigger point.
856  * if delayCount < readCount we return (readCount - delayCount) of
857  * samples from before the trigger fired.
858  */
859  case SUMP_SET_READ_DELAY_COUNT: {
860  printLog("->SUMP_SET_READ_DELAY_COUNT");
861  Sump4ByteComandArg cmd = getSump4ByteComandArg();
862  la_state.read_count = min((uint32_t)cmd.get16(0), la_state.max_capture_size);
863  la_state.delay_count = min((uint32_t)cmd.get16(1),la_state.max_capture_size);
864  printLog("--> read_count: %d", la_state.read_count);
865  printLog("--> delay_count: %d", la_state.delay_count);
866  raiseEvent(READ_DLEAY_COUNT);
867  }
868  break;
869 
870  /* read the rest of the command bytes and check if RLE is enabled. */
871  case SUMP_SET_FLAGS: {
872  printLog("->SUMP_SET_FLAGS");
873  Sump4ByteComandArg cmd = getSump4ByteComandArg();
874  la_state.is_continuous_capture = ((cmd.getPtr()[1] & 0B1000000) != 0);
875  printLog("--> is_continuous_capture: %d\n", la_state.is_continuous_capture);
876  raiseEvent(FLAGS);
877 
878  }
879  break;
880 
881  /* ignore any unrecognized bytes. */
882  default:
883  printLog("->UNHANDLED command: %d", cmd);
884  break;
885 
886  };
887  }
