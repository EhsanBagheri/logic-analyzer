428  void capture(bool is_max_speed) {
429  log("capture is_max_speed: %s", is_max_speed ? "true":"false");
430  // waiting for trigger
431  if (la_state.trigger_mask) {
432  log("waiting for trigger");
433  while ((la_state.trigger_values ^ captureSample()) & la_state.trigger_mask)
434  ;
435  }
436  la_state.setStatus(TRIGGERED);
437  log("triggered");
438 
439  // remove unnecessary entries from buffer based on delayCount & readCount
440  long keep = la_state.read_count - la_state.delay_count;
441  if (keep > 0 && buffer_ptr->available()>keep) {
442  log("keeping last %ld entries",keep);
443  buffer_ptr->clear(buffer_ptr->available() - keep);
444  } else if (keep < 0) {
445  log("ignoring first %ld entries",abs(keep));
446  buffer_ptr->clear(buffer_ptr->available() + abs(keep));
447  } else if (keep==0l){
448  log("starting with clean buffer");
449  buffer_ptr->clear();
450  }
451 
452  // Start Capture
453  if (is_max_speed){
454  if (la_state.is_continuous_capture){
455  captureAllContinousMaxSpeed();
456  } else {
457  captureAllMaxSpeed();
458  dumpData();
459  log("capture-done: %lu",buffer_ptr->available());
460  setStatus(STOPPED);
461  }
462  } else {
463  if (la_state.is_continuous_capture){
464  captureAllContinous();
465  } else {
466  captureAll();
467  dumpData();
468  log("capture-done: %lu",buffer_ptr->available());
469  setStatus(STOPPED);
470  }
471  }

855  void processCommand(int cmd){
856 
857  switch (cmd) {
861  case SUMP_RESET:
862  // debounce reset
863  if (millis()>sump_reset_igorne_timeout){
864  log("=>SUMP_RESET");
865  setStatus(STOPPED);
866  clear();
867  sump_reset_igorne_timeout = millis()+ 500;
868  raiseEvent(RESET);
869  }
870  break;
871 
875  case SUMP_ID:
876  log("=>SUMP_ID");
877  stream().write(device_id, 4);
878  stream().flush();
879  break;
880 
881  /*
882  * We return a description of our capabilities.
883  * Check the function's comments below.
884  */
885  case SUMP_GET_METADATA:
886  log("=>SUMP_GET_METADATA");
887  sendMetadata();
888  break;
889 
890  /*
891  * Captures the data
892  */
893  case SUMP_ARM:
894  log("=>SUMP_ARM");
895  // clear current data
896  clear();
897  setStatus(ARMED);
898  if (is_capture_on_arm){
899  capture();
900  }
901  break;
902 
903  /*
904  * the trigger mask byte has a '1' for each enabled trigger so
905  * we can just use it directly as our trigger mask.
906  */
907  case SUMP_TRIGGER_MASK:
908  log("=>SUMP_TRIGGER_MASK");
909  setTriggerMask(commandExtPinBitArray());
910  break;
911 
912  /*
913  * trigger_values can be used directly as the value of each bit
914  * defines whether we're looking for it to be high or low.
915  */
916  case SUMP_TRIGGER_VALUES:
917  log("=>SUMP_TRIGGER_VALUES");
918  setTriggerValues(commandExtPinBitArray());
919  break;
920 
921 
922  /* read the rest of the command bytes but ignore them. */
923  case SUMP_TRIGGER_CONFIG:
924  log("=>SUMP_TRIGGER_CONFIG");
925  commandExt();
926  break;
927 
928  /*
929  * the shifting needs to be done on the 32bit unsigned long variable
930  * so that << 16 doesn't end up as zero.
931  */
932  case SUMP_SET_DIVIDER: {
933  log("=>SUMP_SET_DIVIDER");
934  Sump4ByteComandArg cmd = commandExt();
935  uint32_t divider = cmd.get32();
936  log("-divider: %lu\n", divider);
937  setupDelay(divider);
938  }
939  break;
940 
941  /*
942  * this just sets up how many samples there should be before
943  * and after the trigger fires. The readCount is total samples
944  * to return and delayCount number of samples after the trigger.
945  * this sets the buffer splits like 0/100, 25/75, 50/50
946  * for example if readCount == delayCount then we should
947  * return all samples starting from the trigger point.
948  * if delayCount < readCount we return (readCount - delayCount) of
949  * samples from before the trigger fired.
950  */
951  case SUMP_SET_READ_DELAY_COUNT: {
952  Sump4ByteComandArg cmd = commandExt();
