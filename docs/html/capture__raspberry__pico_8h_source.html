139  void logic_analyser_init() {
140  log("Init trigger");
141  // Load a program to capture n pins. This is just a single `in pins, n`
142  // instruction with a wrap.
143  uint16_t capture_prog_instr = pio_encode_in(pio_pins, pin_count);
144  struct pio_program capture_prog = {
145  .instructions = &capture_prog_instr,
146  .length = 1,
147  .origin = -1
148  };
149  uint offset = pio_add_program(pio, &capture_prog);
150 
151  // Configure state machine to loop over this `in` instruction forever,
152  // with autopush enabled.
153  pio_sm_config c = pio_get_default_sm_config();
154  sm_config_set_in_pins(&c, pin_base);
155  sm_config_set_wrap(&c, offset, offset);
156  sm_config_set_clkdiv(&c, divider_value);
157  // Note that we may push at a < 32 bit threshold if pin_count does not
158  // divide 32. We are using shift-to-right, so the sample data ends up
159  // left-justified in the FIFO in this case, with some zeroes at the LSBs.
160  sm_config_set_in_shift(&c, true, true, bit_count());
161  sm_config_set_fifo_join(&c, PIO_FIFO_JOIN_RX);
162  pio_sm_init(pio, sm, offset, &c);
163  }
