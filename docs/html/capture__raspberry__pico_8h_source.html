102  void setupTestSignal() {

104  gpio_set_function(pin_base, GPIO_FUNC_PWM);
105  gpio_set_function(pin_base + 1, GPIO_FUNC_PWM);
106  // Topmost value of 3: count from 0 to 3 and then wrap, so period is 4 cycles
107  pwm_hw->slice[0].top = 3;
108  // Divide frequency by two to slow things down a little
109  pwm_hw->slice[0].div = 4 << PWM_CH0_DIV_INT_LSB;
110  // Set channel A to be high for 1 cycle each period (duty cycle 1/4) and
111  // channel B for 3 cycles (duty cycle 3/4)
112  pwm_hw->slice[0].cc =
113  (1 << PWM_CH0_CC_A_LSB) |
114  (3 << PWM_CH0_CC_B_LSB);

129  void arm() {
130  log("Init trigger");
131 
132  // Grant high bus priority to the DMA, so it can shove the processors out
133  // of the way. This should only be needed if you are pushing things up to
134  // >16bits/clk here, i.e. if you need to saturate the bus completely.
135  bus_ctrl_hw->priority = BUSCTRL_BUS_PRIORITY_DMA_W_BITS | BUSCTRL_BUS_PRIORITY_DMA_R_BITS;
136 
137  // Load a program to capture n pins. This is just a single `in pins, n`
138  // instruction with a wrap.
139  uint16_t capture_prog_instr = pio_encode_in(pio_pins, pin_count);
140  struct pio_program capture_prog = {
141  .instructions = &capture_prog_instr,
142  .length = 1,
143  .origin = -1
144  };
145  uint offset = pio_add_program(pio, &capture_prog);
146 
147  // Configure state machine to loop over this `in` instruction forever,
148  // with autopush enabled.
149  pio_sm_config c = pio_get_default_sm_config();
150  sm_config_set_in_pins(&c, pin_base);
151  sm_config_set_wrap(&c, offset, offset);
152  sm_config_set_clkdiv(&c, divider_value);
153  // Note that we may push at a < 32 bit threshold if pin_count does not
154  // divide 32. We are using shift-to-right, so the sample data ends up
155  // left-justified in the FIFO in this case, with some zeroes at the LSBs.
156  sm_config_set_in_shift(&c, true, true, bit_count());
157  sm_config_set_fifo_join(&c, PIO_FIFO_JOIN_RX);
158  pio_sm_init(pio, sm, offset, &c);
159 
161  log("Arming trigger");
162  pio_sm_set_enabled(pio, sm, false);
163  // Need to clear _input shift counter_, as well as FIFO, because there may be
164  // partial ISR contents left over from a previous run. sm_restart does this.
165  pio_sm_clear_fifos(pio, sm);
166  pio_sm_restart(pio, sm);
167 
168  dma_channel_config dma_config = dma_channel_get_default_config(dma_chan);
169  channel_config_set_read_increment(&dma_config, false);
170  channel_config_set_write_increment(&dma_config, true);
171  channel_config_set_transfer_data_size(&dma_config, DMA_SIZE_32);
172  channel_config_set_dreq(&dma_config, pio_get_dreq(pio, sm, false));
173 
174  dma_channel_configure(dma_chan, &dma_config,
175  logicAnalyzer().buffer().data_ptr(), // Destination pointer
176  &pio->rxf[sm], // Source pointer
177  n_samples /4 * sizeof(PinBitArray), // Number of transfers
178  true // Start immediately
179  );
180 
183  start_time = micros();
184  pio_sm_set_enabled(pio, sm, true);
185  }
